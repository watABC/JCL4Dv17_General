  //JCL_lst_Make_Join_byCode  //20191202 wat  //別テーブルから名前をジョイン、キーはコード（文字列）C_POINTER($1;$tblPtr)$tblPtr:=$1  //テーブルポインタC_POINTER($2;$codeFldPtr)$codeFldPtr:=$2  //コードフィールドのポインタC_POINTER($3;$NameFldPtr)$NameFldPtr:=$3  //名前フィールドのポインタC_POINTER($4;$codeAryPtr)$codeAryPtr:=$4  //コードの配列C_POINTER($5;$NameAryPtr)$NameAryPtr:=$5  //返す、名前の配列ARRAY TEXT($aryCode;0)ARRAY TEXT($aryName;0)C_LONGINT($i;$sizeOfAry)  //キャッシュ対象のテーブルからIDと名前を配列に取得READ ONLY($tblPtr->)ALL RECORDS($tblPtr->)FIRST RECORD($tblPtr->)SELECTION TO ARRAY($codeFldPtr->;$aryCode)SELECTION TO ARRAY($NameFldPtr->;$aryName)$sizeOfAry:=Size of array($codeAryPtr->)For ($i;1;$sizeOfAry)		$pos:=Find in array($aryCode;$codeAryPtr->{$i})	If ($pos>0)				$NameAryPtr->{$i}:=$aryName{$pos}			Else 		$NameAryPtr->{$i}:=""			End if 	End for
