  //ファイルパスを生成

C_TEXT($1;$folder)
$folder:=$1  //フォルダ名
C_TEXT($2;$fileName)
$fileName:=$2  //ファイル名
C_TEXT($0;$outPath)

C_TEXT($separator)
$separator:=JCL_file_DirSeparator 

C_LONGINT($length)
C_TEXT($char)

$length:=Length($folder)

  //末尾の１文字取得
$char:=Substring($folder;$length)

If ($char=$separator)
	  //セパレータならそのまま結合
	$outPath:=$folder+$fileName
Else 
	  //セパレータじゃないなら、セパレータを入れて結合
	$outPath:=$folder+$separator+$fileName
End if 

$0:=$outPath
