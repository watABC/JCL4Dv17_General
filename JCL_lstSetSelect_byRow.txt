  //JCL_lstSetSelect_byRow  //20191102 wat  //リストボックスを行番号で選択  //行番号がゼロの場合は何も選択しない、という仕様にした。C_TEXT($1;$inListBox)$inListBox:=$1  //対象となるリストボックス名C_LONGINT($2;$row_number)$row_number:=$2  //行番号C_POINTER($objPtr)$objPtr:=OBJECT Get pointer(Object named;$inListBox)  //行番号がゼロの場合は何も選択しないIf ($row_number#0)	LISTBOX SELECT ROW($objPtr->;$row_number;lk replace selection)	Else 	  //ゼロの場合はデセレクト	LISTBOX SELECT ROW($objPtr->;0;lk remove from selection)	End if
