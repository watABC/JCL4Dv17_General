  //JCL_lst_SortMulti_Acsend  //20191213 wat  //昇順のみサポート、配列要素は２または３、印刷のブレーク処理のために実装  // ソート履歴を実行 XX01_lstXX_makeで実行するJCL_lst_Sortとは異なり、してください  // 「ソート履歴」配列に２回か３回要素をアペンドしていることが前提  // ソート履歴は、呼ぶ直前に指定する。  //マルチレベルソート、4段階まで対応C_TEXT($1;$lstboxName)$lstboxName:=$1C_POINTER($2;$aryHeaderNamesPtr)$aryHeaderNamesPtr:=$2  //C_POINTER($3;$arySortOrdersPtr)  //$arySortOrdersPtr:=$3C_LONGINT($i;$sizeOfAry)C_LONGINT($nr1;$nr2;$nr3;$nr4)$sizeOfAry:=Size of array($aryHeaderNamesPtr->)If ($sizeOfAry#0)	Case of 		: ($sizeOfAry=2)			  //昇順のみサポート、配列要素は２			$nr1:=JCL_lst_ColNumber ($lstboxName;$aryHeaderNamesPtr->{1})			$nr2:=JCL_lst_ColNumber ($lstboxName;$aryHeaderNamesPtr->{2})						  //昇順			LISTBOX SORT COLUMNS(*;$lstboxName;$nr1;>;$nr2;>)					: ($sizeOfAry=3)			  //昇順のみサポート、配列要素は３			$nr1:=JCL_lst_ColNumber ($lstboxName;$aryHeaderNamesPtr->{1})			$nr2:=JCL_lst_ColNumber ($lstboxName;$aryHeaderNamesPtr->{2})			$nr3:=JCL_lst_ColNumber ($lstboxName;$aryHeaderNamesPtr->{3})						  //昇順			LISTBOX SORT COLUMNS(*;$lstboxName;$nr1;>;$nr2;>;$nr3;>)					: ($sizeOfAry=4)			  //昇順のみサポート、配列要素は４//20200304 ike 作りかけだったのを直した			$nr1:=JCL_lst_ColNumber ($lstboxName;$aryHeaderNamesPtr->{1})			$nr2:=JCL_lst_ColNumber ($lstboxName;$aryHeaderNamesPtr->{2})			$nr3:=JCL_lst_ColNumber ($lstboxName;$aryHeaderNamesPtr->{3})			$nr4:=JCL_lst_ColNumber ($lstboxName;$aryHeaderNamesPtr->{4})						  //昇順			LISTBOX SORT COLUMNS(*;$lstboxName;$nr1;>;$nr2;>;$nr3;>;$nr4;>)				End case 	End if
